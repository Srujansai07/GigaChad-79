```jsx
// BroGigaChat MVP - Full Prototype
import React, { useState, useEffect } from 'react';
import { Flame, Trophy, Zap, Clock, Target, Users, Settings, Plus, ChevronRight, X, Check, AlertCircle, TrendingUp, Crown, Shield } from 'lucide-react';

const BroGigaChat = () => {
  // Core state management with persistence
  const [user, setUser] = useState({
    username: 'NewGrinder',
    aura: 0,
    level: 1,
    streak: 0,
    lastActive: new Date().toISOString(),
    rank: 0,
    tasksCompleted: 0,
    strictModeCount: 0
  });

  const [tasks, setTasks] = useState([]);
  const [activeNotification, setActiveNotification] = useState(null);
  const [currentScreen, setCurrentScreen] = useState('home');
  const [showAddTask, setShowAddTask] = useState(false);
  const [strictModeActive, setStrictModeActive] = useState(false);
  const [showOnboarding, setShowOnboarding] = useState(true);
  const [skipCount, setSkipCount] = useState({});

  // Initialize from storage
  useEffect(() => {
    const loadData = async () => {
      try {
        const userData = await window.storage.get('bro_user');
        const tasksData = await window.storage.get('bro_tasks');
        const skipData = await window.storage.get('bro_skips');
        const onboardingDone = await window.storage.get('bro_onboarding');
        
        if (userData?.value) setUser(JSON.parse(userData.value));
        if (tasksData?.value) setTasks(JSON.parse(tasksData.value));
        if (skipData?.value) setSkipCount(JSON.parse(skipData.value));
        if (onboardingDone?.value) setShowOnboarding(false);
      } catch (e) {
        console.log('First time user');
      }
    };
    loadData();
  }, []);

  // Save to storage
  useEffect(() => {
    const saveData = async () => {
      try {
        await window.storage.set('bro_user', JSON.stringify(user));
        await window.storage.set('bro_tasks', JSON.stringify(tasks));
        await window.storage.set('bro_skips', JSON.stringify(skipCount));
      } catch (e) {
        console.log('Storage error:', e);
      }
    };
    if (!showOnboarding) saveData();
  }, [user, tasks, skipCount, showOnboarding]);

  // Calculate user level based on aura
  const getLevel = (aura) => {
    if (aura < 1000) return { level: 1, name: 'Rookie', color: 'text-gray-400' };
    if (aura < 5000) return { level: 2, name: 'Grinder', color: 'text-blue-400' };
    if (aura < 15000) return { level: 3, name: 'Hustler', color: 'text-purple-400' };
    if (aura < 50000) return { level: 4, name: 'Alpha', color: 'text-yellow-400' };
    if (aura < 150000) return { level: 5, name: 'Sigma', color: 'text-red-400' };
    return { level: 6, name: 'TopG', color: 'text-orange-400' };
  };

  const levelInfo = getLevel(user.aura);

  // Add new task
  const addTask = (taskData) => {
    const newTask = {
      id: Date.now(),
      ...taskData,
      created: new Date().toISOString(),
      completed: false,
      skips: 0
    };
    setTasks([...tasks, newTask]);
    setShowAddTask(false);
  };

  // Handle notification actions
  const handleExtend = (taskId, minutes) => {
    const task = tasks.find(t => t.id === taskId);
    const currentSkips = skipCount[taskId] || 0;
    const newSkips = currentSkips + 1;
    
    setSkipCount({ ...skipCount, [taskId]: newSkips });
    
    // Update task with new time
    setTasks(tasks.map(t => 
      t.id === taskId 
        ? { ...t, scheduledFor: new Date(Date.now() + minutes * 60000).toISOString(), skips: newSkips }
        : t
    ));

    // Deduct aura for skipping
    setUser({ ...user, aura: Math.max(0, user.aura - 50) });

    // Check if strict mode should activate
    if (newSkips >= 3) {
      activateStrictMode(task);
    } else {
      setActiveNotification(null);
    }
  };

  const activateStrictMode = (task) => {
    setStrictModeActive(true);
    setActiveNotification(null);
    
    setTimeout(() => {
      completeTask(task.id, true);
      setStrictModeActive(false);
    }, 3000);
  };

  const completeTask = (taskId, fromStrict = false) => {
    const task = tasks.find(t => t.id === taskId);
    const skips = skipCount[taskId] || 0;
    
    // Calculate aura reward
    let auraGain = task.baseAura || 100;
    if (skips === 0) auraGain *= 2; // First notification bonus
    if (fromStrict) auraGain *= 1.5; // Strict mode bonus
    if (user.streak > 7) auraGain *= 1.5; // Streak bonus
    
    auraGain = Math.floor(auraGain);

    // Update user stats
    setUser({
      ...user,
      aura: user.aura + auraGain,
      tasksCompleted: user.tasksCompleted + 1,
      streak: user.streak + 1,
      strictModeCount: fromStrict ? user.strictModeCount + 1 : user.strictModeCount,
      lastActive: new Date().toISOString()
    });

    // Mark task as completed
    setTasks(tasks.map(t => 
      t.id === taskId 
        ? { ...t, completed: true, completedAt: new Date().toISOString(), auraGained: auraGain }
        : t
    ));

    // Clear skip count
    const newSkipCount = { ...skipCount };
    delete newSkipCount[taskId];
    setSkipCount(newSkipCount);

    setActiveNotification(null);

    // Show celebration
    setTimeout(() => {
      if (document.getElementById('celebration')) {
        document.getElementById('celebration').classList.add('animate-ping');
      }
    }, 100);
  };

  // Simulate notification trigger
  const triggerNotification = (task) => {
    setActiveNotification(task);
  };

  // Mock leaderboard data
  const leaderboard = [
    { rank: 1, username: 'SigmaGrinder', aura: 187000, level: 'TopG' },
    { rank: 2, username: 'HustleKing', aura: 156000, level: 'TopG' },
    { rank: 3, username: 'DisciplinedAF', aura: 134000, level: 'Sigma' },
    { rank: 4, username: 'AlphaMentality', aura: 89000, level: 'Alpha' },
    { rank: 5, username: 'GrindNeverStops', aura: 67000, level: 'Alpha' },
  ];

  const activeTasks = tasks.filter(t => !t.completed);
  const completedTasks = tasks.filter(t => t.completed);

  // Onboarding Component
  if (showOnboarding) {
    return (
      <div className="min-h-screen bg-black text-white flex items-center justify-center p-4">
        <div className="max-w-md w-full space-y-6 text-center">
          <div className="space-y-2">
            <div className="text-6xl mb-4">üí™</div>
            <h1 className="text-4xl font-bold text-red-500">BroGigaChat</h1>
            <p className="text-gray-400">Your Phone's New Strict Older Brother</p>
          </div>
          
          <div className="space-y-4 text-left bg-gray-900 p-6 rounded-lg">
            <div className="flex items-start gap-3">
              <Zap className="text-yellow-400 mt-1" size={20} />
              <div>
                <h3 className="font-bold">One-Tap to Action</h3>
                <p className="text-sm text-gray-400">Click notification ‚Üí instantly at your task</p>
              </div>
            </div>
            <div className="flex items-start gap-3">
              <Flame className="text-red-400 mt-1" size={20} />
              <div>
                <h3 className="font-bold">Strict Mode</h3>
                <p className="text-sm text-gray-400">Skip 3 times? I'll FORCE you to work</p>
              </div>
            </div>
            <div className="flex items-start gap-3">
              <Trophy className="text-yellow-400 mt-1" size={20} />
              <div>
                <h3 className="font-bold">Aura Points</h3>
                <p className="text-sm text-gray-400">Build status, compete globally</p>
              </div>
            </div>
          </div>

          <input
            type="text"
            placeholder="Enter your username"
            className="w-full bg-gray-900 border border-gray-700 rounded-lg px-4 py-3 focus:border-red-500 focus:outline-none"
            value={user.username}
            onChange={(e) => setUser({ ...user, username: e.target.value })}
          />

          <button
            onClick={async () => {
              await window.storage.set('bro_onboarding', 'done');
              setShowOnboarding(false);
            }}
            className="w-full bg-red-600 hover:bg-red-700 py-3 rounded-lg font-bold transition-colors"
          >
            LET'S GO üî•
          </button>
        </div>
      </div>
    );
  }

  // Main App Interface
  return (
    <div className="min-h-screen bg-black text-white">
      {/* Strict Mode Overlay */}
      {strictModeActive && (
        <div className="fixed inset-0 z-50 bg-red-600 flex items-center justify-center animate-pulse">
          <div className="text-center space-y-4 p-8">
            <AlertCircle size={80} className="mx-auto animate-bounce" />
            <h1 className="text-4xl font-bold">STRICT MODE ACTIVATED</h1>
            <p className="text-xl">You've skipped 3 times.</p>
            <p className="text-lg">No more excuses.</p>
            <p className="text-2xl font-bold mt-4">DO THE WORK NOW.</p>
            <div className="mt-8 text-sm text-red-200">Forcing app open in 3 seconds...</div>
          </div>
        </div>
      )}

      {/* Active Notification */}
      {activeNotification && !strictModeActive && (
        <div className="fixed top-4 left-4 right-4 z-40 bg-gray-900 border-2 border-red-500 rounded-xl p-4 shadow-2xl animate-bounce">
          <div className="space-y-3">
            <div className="flex items-start justify-between">
              <div>
                <div className="text-xs text-red-400 font-bold mb-1">BROGIGACHAT</div>
                <h3 className="font-bold text-lg">{activeNotification.title}</h3>
                <p className="text-sm text-gray-400">{activeNotification.description}</p>
              </div>
              <button onClick={() => setActiveNotification(null)} className="text-gray-500">
                <X size={20} />
              </button>
            </div>
            
            <div className="text-xs text-gray-500">
              Streak: {user.streak} days | Aura: +{activeNotification.baseAura || 100}
            </div>

            {(skipCount[activeNotification.id] || 0) > 0 && (
              <div className="text-sm text-yellow-400 flex items-center gap-2">
                <AlertCircle size={16} />
                Skip #{skipCount[activeNotification.id]} - {3 - (skipCount[activeNotification.id] || 0)} until Strict Mode
              </div>
            )}

            <div className="grid grid-cols-3 gap-2">
              <button
                onClick={() => handleExtend(activeNotification.id, 10)}
                className="bg-gray-800 hover:bg-gray-700 py-2 px-3 rounded-lg text-sm transition-colors"
              >
                +10min
              </button>
              <button
                onClick={() => handleExtend(activeNotification.id, 30)}
                className="bg-gray-800 hover:bg-gray-700 py-2 px-3 rounded-lg text-sm transition-colors"
              >
                +30min
              </button>
              <button
                onClick={() => completeTask(activeNotification.id)}
                className="bg-red-600 hover:bg-red-700 py-2 px-3 rounded-lg text-sm font-bold transition-colors"
              >
                DO IT
              </button>
            </div>
          </div>
        </div>
      )}

      {/* Top Bar */}
      <div className="bg-gray-900 border-b border-gray-800 px-4 py-3">
        <div className="flex items-center justify-between">
          <div className="flex items-center gap-3">
            <div className="w-10 h-10 bg-red-600 rounded-full flex items-center justify-center font-bold">
              {user.username[0].toUpperCase()}
            </div>
            <div>
              <div className="font-bold">{user.username}</div>
              <div className={`text-xs ${levelInfo.color}`}>{levelInfo.name} ‚Ä¢ Lvl {levelInfo.level}</div>
            </div>
          </div>
          <div className="text-right">
            <div className="flex items-center gap-1 text-yellow-400 font-bold text-lg">
              <Trophy size={20} />
              {user.aura.toLocaleString()}
            </div>
            <div className="text-xs text-gray-400">Aura Points</div>
          </div>
        </div>

        {/* Streak Indicator */}
        {user.streak > 0 && (
          <div className="mt-3 flex items-center gap-2 bg-orange-500/20 border border-orange-500/30 rounded-lg px-3 py-2">
            <Flame className="text-orange-400" size={18} />
            <span className="text-sm font-bold">{user.streak} Day Streak üî•</span>
          </div>
        )}
      </div>

      {/* Main Content */}
      <div className="pb-20">
        {currentScreen === 'home' && (
          <div className="p-4 space-y-4">
            {/* Quick Stats */}
            <div className="grid grid-cols-3 gap-2">
              <div className="bg-gray-900 rounded-lg p-3 text-center">
                <div className="text-2xl font-bold text-green-400">{activeTasks.length}</div>
                <div className="text-xs text-gray-400">Active</div>
              </div>
              <div className="bg-gray-900 rounded-lg p-3 text-center">
                <div className="text-2xl font-bold text-blue-400">{user.tasksCompleted}</div>
                <div className="text-xs text-gray-400">Done</div>
              </div>
              <div className="bg-gray-900 rounded-lg p-3 text-center">
                <div className="text-2xl font-bold text-purple-400">{user.strictModeCount}</div>
                <div className="text-xs text-gray-400">Forced</div>
              </div>
            </div>

            {/* Active Tasks */}
            <div>
              <div className="flex items-center justify-between mb-3">
                <h2 className="text-xl font-bold">Your Tasks</h2>
                <button
                  onClick={() => setShowAddTask(true)}
                  className="bg-red-600 hover:bg-red-700 rounded-full p-2 transition-colors"
                >
                  <Plus size={20} />
                </button>
              </div>

              {activeTasks.length === 0 ? (
                <div className="bg-gray-900 rounded-xl p-8 text-center">
                  <Target className="mx-auto mb-3 text-gray-600" size={48} />
                  <p className="text-gray-400 mb-4">No tasks yet. Let's add one!</p>
                  <button
                    onClick={() => setShowAddTask(true)}
                    className="bg-red-600 hover:bg-red-700 px-6 py-2 rounded-lg font-bold transition-colors"
                  >
                    Add Your First Task
                  </button>
                </div>
              ) : (
                <div className="space-y-3">
                  {activeTasks.map(task => (
                    <div key={task.id} className="bg-gray-900 rounded-xl p-4">
                      <div className="flex items-start justify-between mb-2">
                        <div className="flex-1">
                          <h3 className="font-bold">{task.title}</h3>
                          <p className="text-sm text-gray-400">{task.description}</p>
                        </div>
                        <div className="text-right">
                          <div className="text-yellow-400 text-sm font-bold">+{task.baseAura || 100}</div>
                          <div className="text-xs text-gray-500">Aura</div>
                        </div>
                      </div>
                      
                      {task.skips > 0 && (
                        <div className="text-xs text-red-400 mb-2">
                          ‚ö†Ô∏è Skipped {task.skips} time(s)
                        </div>
                      )}

                      <div className="flex gap-2">
                        <button
                          onClick={() => triggerNotification(task)}
                          className="flex-1 bg-gray-800 hover:bg-gray-700 py-2 rounded-lg text-sm transition-colors"
                        >
                          Test Notification
                        </button>
                        <button
                          onClick={() => completeTask(task.id)}
                          className="flex-1 bg-red-600 hover:bg-red-700 py-2 rounded-lg text-sm font-bold transition-colors"
                        >
                          Complete Now
                        </button>
                      </div>
                    </div>
                  ))}
                </div>
              )}
            </div>

            {/* Completed Tasks */}
            {completedTasks.length > 0 && (
              <div>
                <h3 className="text-lg font-bold mb-3 text-green-400">‚úì Completed Today</h3>
                <div className="space-y-2">
                  {completedTasks.slice(0, 3).map(task => (
                    <div key={task.id} className="bg-gray-900/50 rounded-lg p-3 flex items-center justify-between">
                      <div className="flex items-center gap-3">
                        <Check className="text-green-400" size={20} />
                        <div>
                          <div className="font-medium text-sm">{task.title}</div>
                          <div className="text-xs text-gray-500">+{task.auraGained} Aura earned</div>
                        </div>
                      </div>
                    </div>
                  ))}
                </div>
              </div>
            )}
          </div>
        )}

        {currentScreen === 'leaderboard' && (
          <div className="p-4 space-y-4">
            <h2 className="text-2xl font-bold">Global Rankings</h2>
            
            <div className="bg-gradient-to-br from-yellow-600/20 to-orange-600/20 border border-yellow-500/30 rounded-xl p-4">
              <div className="flex items-center justify-between">
                <div>
                  <div className="text-sm text-gray-400">Your Rank</div>
                  <div className="text-3xl font-bold text-yellow-400">Coming Soon</div>
                  <div className="text-sm text-gray-400 mt-1">Keep grinding to unlock rankings!</div>
                </div>
                <Crown className="text-yellow-400" size={48} />
              </div>
            </div>

            <div className="space-y-2">
              {leaderboard.map((player, idx) => (
                <div
                  key={player.rank}
                  className={`rounded-xl p-4 flex items-center justify-between ${
                    idx < 3 ? 'bg-gradient-to-r from-yellow-600/20 to-transparent border border-yellow-500/20' : 'bg-gray-900'
                  }`}
                >
                  <div className="flex items-center gap-3">
                    <div className={`text-2xl font-bold ${
                      idx === 0 ? 'text-yellow-400' : idx === 1 ? 'text-gray-300' : idx === 2 ? 'text-orange-400' : 'text-gray-500'
                    }`}>
                      #{player.rank}
                    </div>

                <Crown className="text-yellow-400" size={48} />
              </div>
            </div>

            <div className="space-y-2">
              {leaderboard.map((player, idx) => (
                <div
                  key={player.rank}
                  className={`rounded-xl p-4 flex items-center justify-between ${
                    idx < 3 ? 'bg-gradient-to-r from-yellow-600/20 to-transparent border border-yellow-500/20' : 'bg-gray-900'
                  }`}
                >
                  <div className="flex items-center gap-3">
                    <div className={`text-2xl font-bold ${
                      idx === 0 ? 'text-yellow-400' : idx === 1 ? 'text-gray-300' : idx === 2 ? 'text-orange-400' : 'text-gray-500'
                    }`}>
                      #{player.rank}
                    </div>
                    <div>
                      <div className="font-bold">{player.username}</div>
                      <div className="text-xs text-gray-400">{player.level}</div>
                    </div>
                  </div>
                  <div className="text-right">
                    <div className="font-bold text-yellow-400">{player.aura.toLocaleString()}</div>
                    <div className="text-xs text-gray-400">Aura</div>
                  </div>
                </div>
              ))}
            </div>
          </div>
        )}

        {currentScreen === 'profile' && (
          <div className="p-4 space-y-4">
            <div className="bg-gradient-to-br from-red-600/20 to-purple-600/20 rounded-xl p-6 text-center">
              <div className="w-20 h-20 bg-red-600 rounded-full flex items-center justify-center font-bold text-3xl mx-auto mb-3">
                {user.username[0].toUpperCase()}
              </div>
              <h2 className="text-2xl font-bold mb-1">{user.username}</h2>
              <div className={`text-lg ${levelInfo.color} font-bold mb-3`}>
                {levelInfo.name} ‚Ä¢ Level {levelInfo.level}
              </div>
              <div className="text-3xl font-bold text-yellow-400">{user.aura.toLocaleString()} Aura</div>
            </div>

            <div className="bg-gray-900 rounded-xl p-4 space-y-3">
              <h3 className="font-bold text-lg mb-3">Stats</h3>
              <div className="flex items-center justify-between">
                <span className="text-gray-400">Tasks Completed</span>
                <span className="font-bold">{user.tasksCompleted}</span>
              </div>
              <div className="flex items-center justify-between">
                <span className="text-gray-400">Current Streak</span>
                <span className="font-bold flex items-center gap-1">
                  <Flame className="text-orange-400" size={16} />
                  {user.streak} days
                </span>
              </div>
              <div className="flex items-center justify-between">
                <span className="text-gray-400">Strict Modes Survived</span>
                <span className="font-bold">{user.strictModeCount}</span>
              </div>
              <div className="flex items-center justify-between">
                <span className="text-gray-400">Success Rate</span>
                <span className="font-bold text-green-400">
                  {user.tasksCompleted > 0 ? Math.round((user.tasksCompleted / (user.tasksCompleted + Object.keys(skipCount).length)) * 100) : 100}%
                </span>
              </div>
            </div>

            <div className="bg-gray-900 rounded-xl p-4">
              <h3 className="font-bold mb-3">Badges</h3>
              <div className="grid grid-cols-3 gap-3">
                <div className="bg-gray-800 rounded-lg p-3 text-center">
                  <Shield className="text-purple-400 mx-auto mb-1" size={24} />
                  <div className="text-xs">Founder</div>
                </div>
                {user.streak >= 7 && (
                  <div className="bg-gray-800 rounded-lg p-3 text-center">
                    <Flame className="text-orange-400 mx-auto mb-1" size={24} />
                    <div className="text-xs">7-Day</div>
                  </div>
                )}
                {user.strictModeCount >= 5 && (
                  <div className="bg-gray-800 rounded-lg p-3 text-center">
                    <Zap className="text-yellow-400 mx-auto mb-1" size={24} />
                    <div className="text-xs">Forced</div>
                  </div>
                )}
              </div>
            </div>

            <button
              onClick={async () => {
                if (confirm('Reset all data? This cannot be undone.')) {
                  await window.storage.delete('bro_user');
                  await window.storage.delete('bro_tasks');
                  await window.storage.delete('bro_skips');
                  await window.storage.delete('bro_onboarding');
                  window.location.reload();
                }
              }}
              className="w-full bg-gray-800 hover:bg-gray-700 py-3 rounded-lg text-sm transition-colors"
            >
              Reset Progress
            </button>
          </div>
        )}
      </div>

      {/* Add Task Modal */}
      {showAddTask && (
        <div className="fixed inset-0 bg-black/90 z-30 flex items-end md:items-center justify-center p-4">
          <div className="bg-gray-900 rounded-t-2xl md:rounded-2xl w-full md:max-w-lg p-6 space-y-4">
            <div className="flex items-center justify-between mb-4">
              <h2 className="text-xl font-bold">Add New Task</h2>
              <button onClick={() => setShowAddTask(false)}>
                <X size={24} />
              </button>
            </div>

            <div className="space-y-4">
              <div>
                <label className="block text-sm font-medium mb-2">What do you want to do?</label>
                <input
                  type="text"
                  placeholder="e.g., Post on Twitter, Go to gym"
                  className="w-full bg-gray-800 border border-gray-700 rounded-lg px-4 py-3 focus:border-red-500 focus:outline-none"
                  value={newTask.title}
                  onChange={(e) => setNewTask({ ...newTask, title: e.target.value })}
                />
              </div>

              <div>
                <label className="block text-sm font-medium mb-2">Details (optional)</label>
                <input
                  type="text"
                  placeholder="e.g., Tweet about AI trends"
                  className="w-full bg-gray-800 border border-gray-700 rounded-lg px-4 py-3 focus:border-red-500 focus:outline-none"
                  value={newTask.description}
                  onChange={(e) => setNewTask({ ...newTask, description: e.target.value })}
                />
              </div>

              <div>
                <label className="block text-sm font-medium mb-2">Which app?</label>
                <select
                  className="w-full bg-gray-800 border border-gray-700 rounded-lg px-4 py-3 focus:border-red-500 focus:outline-none"
                  value={newTask.app}
                  onChange={(e) => setNewTask({ ...newTask, app: e.target.value })}
                >
                  <option>Twitter</option>
                  <option>Instagram</option>
                  <option>YouTube</option>
                  <option>Gym App</option>
                  <option>Notes</option>
                  <option>Other</option>
                </select>
              </div>

              <button
                onClick={addTask}
                disabled={!newTask.title}
                className="w-full bg-red-600 hover:bg-red-700 disabled:bg-gray-700 disabled:cursor-not-allowed py-3 rounded-lg font-bold transition-colors"
              >
                Add Task
              </button>
            </div>
          </div>
        </div>
      )}

      {/* Bottom Navigation */}
      <div className="fixed bottom-0 left-0 right-0 bg-gray-900 border-t border-gray-800">
        <div className="grid grid-cols-3 gap-1 p-2">
          <button
            onClick={() => setCurrentScreen('home')}
            className={`flex flex-col items-center gap-1 py-2 rounded-lg transition-colors ${
              currentScreen === 'home' ? 'bg-red-600 text-white' : 'text-gray-400'
            }`}
          >
            <Target size={24} />
            <span className="text-xs">Tasks</span>
          </button>
          <button
            onClick={() => setCurrentScreen('leaderboard')}
            className={`flex flex-col items-center gap-1 py-2 rounded-lg transition-colors ${
              currentScreen === 'leaderboard' ? 'bg-red-600 text-white' : 'text-gray-400'
            }`}
          >
            <Trophy size={24} />
            <span className="text-xs">Ranks</span>
          </button>
          <button
            onClick={() => setCurrentScreen('profile')}
            className={`flex flex-col items-center gap-1 py-2 rounded-lg transition-colors ${
              currentScreen === 'profile' ? 'bg-red-600 text-white' : 'text-gray-400'
            }`}
          >
            <Users size={24} />
            <span className="text-xs">Profile</span>
          </button>
        </div>
      </div>
    </div>
  );
};

export default BroGigaChat;
```
